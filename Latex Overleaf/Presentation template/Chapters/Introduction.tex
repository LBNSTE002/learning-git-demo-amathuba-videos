\begin{frame}{Including equations/mathematics}
Numbered:
\begin{equation}
    \hat{y}_i = \hat{\beta}_0 + \hat{\beta}_1 x_{1i} + \hat{\beta}_{2} x_{2i} +...+ \hat{\beta}_p x_{pi}
\end{equation}
or not numbered:
\begin{equation*}
    \hat{y}_i = \hat{\beta}_0 + \hat{\beta}_1 x_{1i} + \hat{\beta}_{2} x_{2i} +...+ \hat{\beta}_p x_{pi}
\end{equation*}

You can also include maths in-line (for example: $\hat{\beta_1} = 0.53$) using dollar signs.

\end{frame}
